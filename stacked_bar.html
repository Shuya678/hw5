import sys, subprocess pkgs = ["pandas", "numpy", "plotly", "requests", "beautifulsoup4", "lxml", "country_converter"] subprocess.check_call([sys.executable, "-m", "pip", "install", "-q"] + pkgs) print("Installed OK") subprocess.check_call([sys.executable, "-m", "pip", "install", "-U", "nbformat", "ipython"]) import pandas as pd import plotly.express as px import requests import country_converter as coco # --- 1) scrape GDP table (IMF) safely (avoid 403) --- wiki_url = "https://en.wikipedia.org/wiki/List_of_countries_by_GDP_(nominal)" headers = { "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 " "(KHTML, like Gecko) Chrome/120.0 Safari/537.36" } html = requests.get(wiki_url, headers=headers, timeout=30).text tables = pd.read_html(html) # --- 2) find a table that contains Country + IMF column --- gdp = None for t in tables: cols = [str(c).lower() for c in t.columns] if any("country" in c for c in cols) and any("imf" in c for c in cols): gdp = t.copy() break if gdp is None: raise RuntimeError("No GDP table with IMF column found. Wikipedia table layout may have changed.") # --- 3) normalize columns --- gdp.columns = [str(c).strip() for c in gdp.columns] country_col = [c for c in gdp.columns if "country" in str(c).lower()][0] imf_col = [c for c in gdp.columns if "imf" in str(c).lower()][0] df = gdp[[country_col, imf_col]].rename(columns={country_col: "country", imf_col: "gdp_imf_usd_million"}) # --- 4) clean numbers --- df["country"] = df["country"].astype(str).str.replace(r"\[.*?\]", "", regex=True).str.strip() df["gdp_imf_usd_million"] = ( df["gdp_imf_usd_million"] .astype(str) .str.replace(",", "", regex=False) .str.replace(r"\[.*?\]", "", regex=True) .str.strip() ) df["gdp_imf_usd_million"] = pd.to_numeric(df["gdp_imf_usd_million"], errors="coerce") df = df.dropna(subset=["gdp_imf_usd_million"]) # --- 5) assign region (continent) --- cc = coco.CountryConverter() df["region"] = cc.convert(names=df["country"].tolist(), to="continent") df = df[df["region"].notna() & (df["region"] != "not found")] # --- 6) limit complexity: top N countries per region --- TOP_N = 8 df = df.sort_values(["region", "gdp_imf_usd_million"], ascending=[True, False]) df = df.groupby("region", group_keys=False).head(TOP_N) # --- 7) plot stacked bar --- fig = px.bar( df, x="region", y="gdp_imf_usd_million", color="country", title=f"GDP by Country (IMF) — Top {TOP_N} per Region", labels={"gdp_imf_usd_million": "GDP (IMF, US$ million)", "region": "Region"}, ) fig.update_layout(barmode="stack") fig.show() # --- 8) export html --- fig.write_html("stacked_bar.html", include_plotlyjs="cdn") print("Saved: stacked_bar.html")
